select `grupos`.`gru_des` AS `gru_des`,`proveedores`.`Prv_des` AS `Prv_des`,`productos`.`pro_Des` AS `pro_Des`,`productos`.`Pro_uni` AS `Pro_uni`,`productos`.`pro_cod` AS `pro_cod`,`productos`.`pro_pve` AS `pro_pve`,`productos`.`pro_exi` AS `pro_exi`,`productos`.`pro_ppp` AS `pro_ppp`,`productos`.`pro_PrC` AS `pro_PrC` from ((`productos` join `proveedores` on((`proveedores`.`Prv_id` = `productos`.`prv_id`))) join `grupos` on((`grupos`.`gru_id` = `productos`.`gru_id`)))
